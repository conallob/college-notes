class ROOT_PARTITION_ARR
 
create 
      make
 
feature 
 
      v: ARRAY [STRING];
 
      make is
           local
                 ps: PARTITION_ARR [STRING];
                 pivot: STRING;
           do
                 file2array ("in.txt");
                 print ("Input Sequence is %N");
                 print_array (v, v.lower, v.upper);
                 create ps;
 
                 print ("Choose pivot: Choose number between ");
                 io.put_integer (v.lower);
                 print (" and ");
                 io.put_integer (v.upper);
                 io.new_line;
 
                 io.read_integer;
                 pivot := v.item (io.last_integer);
                 ps.find_partition (v, v.lower, v.upper, pivot);
 
                 print ("Partition about ");
                 print (pivot);
                 io.new_line;
 
                 print_array (v, v.lower, v.upper);
                 io.new_line;
 
                 print ("L val = ");
                 io.put_integer (ps.l);
                 print ("%NR val = ");
                 io.put_integer (ps.r);
 
                 print ("%N%Nexchanges =");
                 io.put_integer (ps.exchs);
                 io.new_line;
 
                 print ("comparisons = ");
                 io.put_integer (ps.comps);
                 io.new_line;
 
                 io.readchar;
                 io.readchar
           end;
 
      file2array (fname: STRING) is
           local
                 in_file: PLAIN_TEXT_FILE;
                 k: INTEGER;
                 t: STRING
           do
                 from
                      create in_file.make_open_read (fname);
                      in_file.read_integer;
                      create v.make (1, in_file.last_integer);
                      k := v.lower
                 until
                      k > v.upper
                 loop
                      in_file.readword;
                      t := clone (in_file.last_string);
                      v.put (t, k);
                      k := k + 1
                 end;
                 in_file.close
           end;
 
      print_array (a0: ARRAY [STRING]; low, high: INTEGER) is
           local
                 i: INTEGER
           do
                 from
                      i := low
                 until
                      i > high
                 loop
                      io.put_string (a0.item (i));
                      if i /= high then
                            io.put_string (",  ")
                      end;
                      i := i + 1
                 end;
                 io.new_line
           end;
      
end -- class ROOT_PARTITION_ARR
