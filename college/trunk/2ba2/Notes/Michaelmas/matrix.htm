class
	MATRIX [G -> COMPARABLE]

creation 
	make

feature {NONE} 

	mat: ARRAY [VECTOR [G]];
			
feature  

	rows: INTEGER;

			
	cols: INTEGER;		

			

	make (r, c: INTEGER) is
		local
			i: INTEGER;
			v: VECTOR [G]
		do
			rows := r;
			cols := c;
			!! mat.make (1, rows);
			from
				i := 1
			until
				i > rows
			loop
				!! v.make (cols);
				mat.put (v, i);
				i := i + 1
			end
		end;

	
	item (i, j: INTEGER): G is
		do
			Result := mat.item (i).item (j)
		end;

	
	put (x: G; i, j: INTEGER) is
		do
			mat.item (i).put (x, j)
		end;

	
	put_row (a: ARRAY [G]; i: INTEGER) is
		local
			v: VECTOR [G]
		do
			!! v.make (cols);
			v.arr_copy (a);
			mat.put (v, i)
		end;

	
	min_row: VECTOR [G] is
		local
			i, index: INTEGER;
			tmin: G;
			r: VECTOR [G]
		do
			from
				i := 1;
				!! Result.make (rows)
			until
				i > rows
			loop
				r := mat.item (i);
				index := r.min_index;
				tmin := r.item (index);
				Result.put (tmin, i);
				i := i + 1
			end
		end;

	
	max_row: VECTOR [G] is
		local
			i, index: INTEGER;
			tmax: G;
			r: VECTOR [G]
		do
			from
				i := 1;
				!! Result.make (rows)
			until
				i > rows
			loop
				r := mat.item (i);
				index := r.max_index;
				tmax := r.item (index);
				Result.put (tmax, i);
				i := i + 1
			end
		end;

	
	transpose: like Current is
		local
			i, j: INTEGER;
			it: G
		do
			!! Result.make (cols, rows);
			from
				i := 1
			until
				i > cols
			loop
				from
					j := 1
				until
					j > rows
				loop
					it := item (j, i);
					Result.put (it, i, j);
					j := j + 1
				end;
				i := i + 1
			end
		end;
	
end -- class MATRIX
