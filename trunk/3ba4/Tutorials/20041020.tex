\begin{eqnarray*}
g & \equiv & r9 \\
i & \equiv & r26 \\
j & \equiv & r27 \\
k & \equiv & r16 \\
\end{eqnarray*}

\begin{verbatim}

	add 	r0,#256,r9		; g = 256

p:

	add 	r26,r27,r16		; k = i + j
	sll 	r16,#2,r1		; r1 = k << 2
	sub 	r1,#1,r1			; r1 = (k << 2) - 1
	ret 	r25,0				; r25 used to return trsult

q:

	add 	r0,r9,r10		; setup param 1
	sub 	r0,r26,r11		; setup param 2
	callr r25,p				; result in r1
	ret	r25,0				; return
	
\end{verbatim}
